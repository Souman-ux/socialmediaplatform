DROP TABLE IF EXISTS Users CASCADE;
DROP TABLE IF EXISTS Followers CASCADE;
DROP TABLE IF EXISTS Comments CASCADE;
DROP TABLE IF EXISTS Posts CASCADE;
DROP TABLE IF EXISTS Messages CASCADE;
DROP TABLE IF EXISTS Friendships CASCADE;





CREATE TABLE IF NOT EXISTS Users (
    user_id SERIAL PRIMARY KEY,
    name VARCHAR(50)NOT NULL UNIQUE,
    email VARCHAR (100) NOT NULL UNIQUE,
    hashed_password VARCHAR(255) NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);


CREATE TABLE IF NOT EXISTS Followers (
    follower_id INT NOT NULL,
    followed_id INT NOT NULL,
    PRIMARY KEY (follower_id, followed_id),
    FOREIGN KEY (follower_id) REFERENCES Users(user_id) ON DELETE CASCADE,
    FOREIGN KEY (followed_id) REFERENCES Users(user_id) ON DELETE CASCADE
);

CREATE TABLE IF NOT EXISTS Comments (
    comment_id SERIAL PRIMARY KEY,
    post_id INT NOT NULL,
    user_id INT NOT NULL,
    content TEXT NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (post_id) REFERENCES Posts(post_id),
    FOREIGN KEY (user_id) REFERENCES Users(user_id)
);


CREATE TABLE IF NOT EXISTS Posts (
    post_id SERIAL PRIMARY KEY,
    user_id INT NOT NULL,
    content TEXT NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES Users(user_id) ON DELETE CASCADE
);


CREATE TABLE IF NOT EXISTS Messages (
    message_id SERIAL PRIMARY KEY,
    sender_id INT NOT NULL,
    receiver_id INT NOT NULL,
    content TEXT NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (sender_id) REFERENCES Users(user_id) ON DELETE CASCADE,
    FOREIGN KEY (receiver_id) REFERENCES Users(user_id) ON DELETE CASCADE
);



CREATE TABLE IF NOT EXISTS Friendships (
    user1_id INT NOT NULL,
    user2_id INT NOT NULL,
    status friendship_status NOT NULL DEFAULT 'pending',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (user1_id, user2_id),
    FOREIGN KEY (user1_id) REFERENCES Users(user_id) ON DELETE CASCADE,
    FOREIGN KEY (user2_id) REFERENCES Users(user_id) ON DELETE CASCADE
);

-- To Join Comments and Posts
SELECT 
    p.post_id, 
    p.content AS post_content, 
    c.comment_id, 
    c.content AS comment_content, 
    c.created_at AS comment_created_at, 
    u.user_id, 
    u.name AS commenter_name
FROM 
    Posts p
JOIN 
    Comments c ON p.post_id = c.post_id
JOIN 
    Users u ON c.user_id = u.user_id;


-- To Join Users and Friendships
SELECT 
    f.user1_id, 
    u1.name AS user1_name, 
    f.user2_id, 
    u2.name AS user2_name, 
    f.status AS friendship_status, 
    f.created_at AS friendship_created_at
FROM 
    Friendships f
JOIN 
    Users u1 ON f.user1_id = u1.user_id
JOIN 
    Users u2 ON f.user2_id = u2.user_id;


-- To Join Users and Posts
SELECT 
    p.post_id, 
    p.content AS post_content, 
    p.created_at AS post_created_at, 
    u.user_id, 
    u.name AS user_name, 
    u.email AS user_email
FROM 
    Posts p
JOIN 
    Users u ON p.user_id = u.user_id;


-- To Join Users and Followers
SELECT 
    u.name AS followed_user_name, 
    f.follower_id, 
    f.followed_id, 
    uf.name AS follower_name
FROM 
    Followers f
JOIN 
    Users u ON f.followed_id = u.user_id
JOIN 
    Users uf ON f.follower_id = uf.user_id;



-- To Join Users and Messages
SELECT 
    m.message_id, 
    m.content AS message_content, 
    m.created_at AS message_created_at, 
    u1.name AS sender_name, 
    u2.name AS receiver_name
FROM 
    Messages m
JOIN 
    Users u1 ON m.sender_id = u1.user_id
JOIN 
    Users u2 ON m.receiver_id = u2.user_id;





-- To Join Post, Comments and Users
SELECT 
    p.post_id, 
    p.content AS post_content, 
    c.comment_id, 
    c.content AS comment_content, 
    c.created_at AS comment_created_at, 
    u.name AS commenter_name
FROM 
    Posts p
JOIN 
    Comments c ON p.post_id = c.post_id
JOIN 
    Users u ON c.user_id = u.user_id;




-- To Join Posts, Comments, Messages and Users
SELECT 
    p.post_id, 
    p.content AS post_content, 
    c.comment_id, 
    c.content AS comment_content, 
    m.message_id, 
    m.content AS message_content, 
    u1.name AS post_user, 
    u2.name AS commenter_name, 
    u3.name AS message_sender_name, 
    u4.name AS message_receiver_name
FROM 
    Posts p
JOIN 
    Comments c ON p.post_id = c.post_id
JOIN 
    Messages m ON p.user_id = m.sender_id
JOIN 
    Users u1 ON p.user_id = u1.user_id
JOIN 
    Users u2 ON c.user_id = u2.user_id
JOIN 
    Users u3 ON m.sender_id = u3.user_id
JOIN 
    Users u4 ON m.receiver_id = u4.user_id;




SELECT 
    p.post_id, 
    p.content AS post_content, 
    p.created_at AS post_created_at, 
    u.name AS user_name
FROM 
    Posts p
JOIN 
    Users u ON p.user_id = u.user_id
WHERE 
    u.user_id = 1;
